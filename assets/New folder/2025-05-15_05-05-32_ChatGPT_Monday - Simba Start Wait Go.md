```tsx
"use client";

import {

export interface AuthContextType {
  userId: string | null;
  orgId: string | null;
  orgRole: string | null;
  isSignedIn: boolean;
  isOrgSelected: boolean;
  redirectToSignIn: (url?: string) => void;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider = ({ children }: { children: React.ReactNode }) => {
  const { user, isSignedIn } = useUser();

  const { signOut } = useClerk();
  const router = useRouter();

 const orgId = organization?.id ?? null;
  const userId = user?.id ?? null;
  const isOrgSelected = Boolean(orgId);

 const orgRole =
    user?.organizationMemberships?.find((m) => m.organization.id === orgId)
      ?.role ?? null;

 const redirectToSignIn = (redirectUrl?: string) => {
    const url = redirectUrl
      ? `/sign-in?redirect_url=${encodeURIComponent(redirectUrl)}`
      : "/sign-in";
    router.push(url);
  };

 return (
    <AuthContext.Provider
      value={{
        userId,
        orgId,
        orgRole,
        isSignedIn,
        isOrgSelected,
        redirectToSignIn,
```

### 🧩 `middleware.ts` – Best-Practices, Modern Clerk Flow

```ts
import { clerkMiddleware, redirectToSignIn } from "@clerk/nextjs/server";
import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";

const PUBLIC_ROUTES = ["/", "/about", "/api/health"];
const STATIC_PATHS = ["/favicon.ico", "/robots.txt", "/_next", "/public"];
const ONBOARDING_ROUTE = "/onboarding";

export default clerkMiddleware(async (auth, req: NextRequest) => {
  const { userId, sessionClaims } = auth();
  const url = req.nextUrl.clone();
  const path = url.pathname;

 const isPublic = PUBLIC_ROUTES.some((route) => path.startsWith(route));
  const isStatic = STATIC_PATHS.some((route) => path.startsWith(route));

 if (isStatic || isPublic) return NextResponse.next();

 // If no user, redirect
  if (!userId) {
    return redirectToSignIn({ returnBackUrl: req.url });
  }

 // Onboarding logic
  const hasOnboarded = sessionClaims?.publicMetadata?.onboarded === true;
  if (!hasOnboarded && !path.startsWith(ONBOARDING_ROUTE)) {
    url.pathname = ONBOARDING_ROUTE;
    return NextResponse.redirect(url);
  }

 return NextResponse.next();
});

export const config = {
  matcher: ["/((?!.*\\..*|_next).*)"],
};
```
